150  virtual void update (TuioTime ttime, float xp, float yp) {
151  TuioPoint lastPoint = path.back();
152  TuioPoint::update(ttime,xp, yp);
153 
154  TuioTime diffTime = currentTime - lastPoint.getTuioTime();
155  float dt = diffTime.getTotalMilliseconds()/1000.0f;
156  float dx = xpos - lastPoint.getX();
157  float dy = ypos - lastPoint.getY();
158  float dist = sqrt(dx*dx+dy*dy);
159  float last_motion_speed = motion_speed;
160 
161  x_speed = dx/dt;
162  y_speed = dy/dt;
163  motion_speed = dist/dt;
164  motion_accel = (motion_speed - last_motion_speed)/dt;
165 
166  TuioPoint p(currentTime,xpos,ypos);
167  path.push_back(p);
168 
169  if (motion_accel>0) state = TUIO_ACCELERATING;
170  else if (motion_accel<0) state = TUIO_DECELERATING;
171  else state = TUIO_STOPPED;
172  };
