52 int Joystick_Reader::Init_Joy() {
53 
54  // first initialise Direct Input or the linux joystick lib
55  if (!InitInput()) {
56  std::cerr << "Failed to initialise the joypad lib\n";
57  return 0;
58  }
59 
60  // make sure there is a joypad attached
61  if (NumJoypads() == 0) {
62  std::cerr << "No Joypad attached\n";
63  return 0;
64  }
65 
66  // make sure there is a joypad attached
67  if (NumJoypads() == 1) {
68  std::cerr << "Only virtual joystick detected\n";
69  return 0;
70  }
71 
72  int joy=1;
73  // get the first available jopypad
74  if( !(g_pJoypad = GetJoypad(joy)) ) {
75  std::cerr << "Failed to access joypad\n";
76  return 0;
77  }
78  return 1;
79 }

81 void Joystick_Reader::Main()
82 {
83 
84  if(!Init_Joy())
85  {
86  LOG(WARNING) << "Joystick device not connected or with wrong permissions.";
87  LOG(ERROR) << PluginDisplayName << " will NOT keep executing";
88  stoped=true;
89  return;
90  }
91  while(running)
92  {
93  UpdateInputState();
94  X_axis=(float)g_pJoypad->Xaxis()/32766;
95  Y_axis=(float)g_pJoypad->Yaxis()/32766;
96  Z_axis=(float)g_pJoypad->Zaxis()/32766;
97  U_axis=(float)g_pJoypad->Xrot()/32766;
98  V_axis=(float)g_pJoypad->Yrot()/32766;
99  W_axis=(float)g_pJoypad->Zrot()/32766;
100  //std::cout << X_axis << " " << Y_axis << " " << Z_axis << " " << RX_axis << " " << RY_axis << " " << RZ_axis << std::endl;
101  }
102  stoped=true;
103  return;
104 }
