209 BOOL CALLBACK EnumObjectsCallback( const DIDEVICEOBJECTINSTANCE* pdidoi,
210  VOID* pContext )
211 {
212  static int nSliderCount = 0; // Number of returned slider controls
213  static int nPOVCount = 0; // Number of returned POV controls
214 
215  // For axes that are returned, set the DIPROP_RANGE property for the
216  // enumerated axis in order to scale min/max values.
217  if( pdidoi->dwType & DIDFT_AXIS )
218  {
219  DIPROPRANGE diprg;
220  diprg.diph.dwSize = sizeof(DIPROPRANGE);
221  diprg.diph.dwHeaderSize = sizeof(DIPROPHEADER);
222  diprg.diph.dwHow = DIPH_BYID;
223  diprg.diph.dwObj = pdidoi->dwType; // Specify the enumerated axis
224  diprg.lMin = -32766;
225  diprg.lMax = +32766;
226 
227  // Set the range for the axis
228  if( FAILED( g_pJoystick->SetProperty( DIPROP_RANGE, &diprg.diph ) ) )
229  return DIENUM_STOP;
230 
231  }
232 
233  // Set the UI to reflect what objects the joystick supports
234  if (pdidoi->guidType == GUID_XAxis)
235  {
236  g_jp->m_iAxes |= XInputJoypad::X_AXIS;
237  }
238  if (pdidoi->guidType == GUID_YAxis)
239  {
240  g_jp->m_iAxes |= XInputJoypad::Y_AXIS;
241  }
242  if (pdidoi->guidType == GUID_ZAxis)
243  {
244  g_jp->m_iAxes |= XInputJoypad::Z_AXIS;
245  }
246  if (pdidoi->guidType == GUID_RxAxis)
247  {
248  g_jp->m_iAxes |= XInputJoypad::X_ROT_AXIS;
249  }
250  if (pdidoi->guidType == GUID_RyAxis)
251  {
252  g_jp->m_iAxes |= XInputJoypad::Y_ROT_AXIS;
253  }
254  if (pdidoi->guidType == GUID_RzAxis)
255  {
256  g_jp->m_iAxes |= XInputJoypad::Z_ROT_AXIS;
257  }
258  if (pdidoi->guidType == GUID_Slider)

264  break;
