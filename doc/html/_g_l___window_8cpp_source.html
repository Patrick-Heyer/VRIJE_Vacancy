44 void CQuaternion::CreateFromAxisAngle(float X, float Y, float Z, float degree)
45 {
46  // This function takes an angle and an axis of rotation, then converts
47  // it to a quaternion. An example of an axis and angle is what we pass into
48  // glRotatef(). That is an axis angle rotation. It is assumed an angle in
49  // degrees is being passed in. Instead of using glRotatef(), we can now handle
50  // the rotations our self.
51 
52  // The equations for axis angle to quaternions are such:
53 
54  // w = cos( theta / 2 )
55  // x = X * sin( theta / 2 )
56  // y = Y * sin( theta / 2 )
57  // z = Z * sin( theta / 2 )
58 
59  // First we want to convert the degrees to radians
60  // since the angle is assumed to be in radians
61  float angle = float((degree / 180.0f) * PI);
62 
63  // Here we calculate the sin( theta / 2) once for optimization
64  float result = (float)sin( angle / 2.0f );
65 
66  // Calcualte the w value by cos( theta / 2 )
67  w = (float)cos( angle / 2.0f );
68 
69  // Calculate the x, y and z of the quaternion
70  x = float(X * result);
71  y = float(Y * result);
72  z = float(Z * result);
73 }

82 void CQuaternion::CreateMatrix(float *pMatrix)
83 {
84  // Make sure the matrix has allocated memory to store the rotation data
85  if(!pMatrix) return;
86 
87  // This function is a necessity when it comes to doing almost anything
88  // with quaternions. Since we are working with OpenGL, which uses a 4x4
89  // homogeneous matrix, we need to have a way to take our quaternion and
90  // convert it to a rotation matrix to modify the current model view matrix.
91  // We pass in a 4x4 matrix, which is a 1D array of 16 floats. This is how OpenGL
92  // allows us to pass in a matrix to glMultMatrixf(), so we use a single dimensioned array.
93  // After about 300 trees murdered and 20 packs of chalk depleted, the
94  // mathematicians came up with these equations for a quaternion to matrix converion:
95  //
96  // Š 2 2 Š
97  // Š 1 - (2y + 2z ) 2xy + 2zw 2xz - 2yw 0 Š
98  // Š Š
99  // Š 2 2 Š
100  // M = Š 2xy - 2zw 1 - (2x + 2z ) 2zy + 2xw 0 Š
101  // Š Š
102  // Š 2 2 Š
103  // Š 2xz + 2yw 2yz - 2xw 1 - (2x + 2y ) 0 Š
104  // Š Š
105  // Š Š
106  // Š 0 0 0 1 | Š
107  // Š Š
108  //
109  // This is of course a 4x4 matrix. Notice that a rotational matrix can just
110  // be a 3x3 matrix, but since OpenGL uses a 4x4 matrix, we need to conform to the man.
111  // Remember that the identity matrix of a 4x4 matrix has a diagonal of 1's, where
112  // the rest of the indices are 0. That is where we get the 0's lining the sides, and
113  // the 1 at the bottom-right corner. Since OpenGL matrices are row by column, we fill
114  // in our matrix accordingly below.
115 
116  // First row
117  pMatrix[ 0] = 1.0f - 2.0f * ( y * y + z * z );
118  pMatrix[ 1] = 2.0f * (x * y + z * w);
119  pMatrix[ 2] = 2.0f * (x * z - y * w);
120  pMatrix[ 3] = 0.0f;
121 
122  // Second row
123  pMatrix[ 4] = 2.0f * ( x * y - z * w );
124  pMatrix[ 5] = 1.0f - 2.0f * ( x * x + z * z );
125  pMatrix[ 6] = 2.0f * (z * y + x * w );
126  pMatrix[ 7] = 0.0f;
127 
128  // Third row
129  pMatrix[ 8] = 2.0f * ( x * z + y * w );
130  pMatrix[ 9] = 2.0f * ( y * z - x * w );
131  pMatrix[10] = 1.0f - 2.0f * ( x * x + y * y );
132  pMatrix[11] = 0.0f;
133 
134  // Fourth row
135  pMatrix[12] = 0;
136  pMatrix[13] = 0;
137  pMatrix[14] = 0;
138  pMatrix[15] = 1.0f;
139 
140  // Now pMatrix[] is a 4x4 homogeneous matrix that can be applied to an OpenGL Matrix
141 }
